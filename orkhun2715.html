 ئافاق خوجا ئەۋلادلىرىنىڭ دەۋاسى


                                                                            ئافاق خوجا ئەۋلادلىرىنىڭ دەۋاسى
                  
                                                                   مۇھەممەت ئىمىن قۇربانى
  
             1935-يىللىرىنىڭ ئاخىرلىرىدا قەشقەر خەلىقىگە غەيرى بىر خەۋەر يىتىپ كەلدى،ئۇ بولسىمۇ شىنجاڭ ئۆلكىلىك ھۆكۈمەتنىڭ بۇيرۇقى ئىدى، بۇيرۇقتا مۇنداق دىيىلگەن ئىدى،«بۇرۇن قەشقەردە ئافاق غوجىنىڭ خىلى كۆپ يەر-زىمىنلىرى بار بولۇپ ئۇنىڭ ھەر يىللىقلىرىدىن ئەۋلاتلىرى بەھرىمان بولۇپ كەلگەن، لېكىن كىيىنكى كۈنلەردە شىئەندە ياشاپ كەلگەن بىرقىسىم ئەۋلاتلىرى شىن خەي ئىنقىلاۋى مەزگىلدە يول توسۇلۇپ قىلىش سەۋەبىدىن 20نەچچە يىللىق ئاشلىقتىن بەھرىمان بولالمىغان، ئۇلار ھازىر تېگىشلىك ھەققىنى تەلەپ قىلىپ مەھكىمىمىزگە ئەرىز قىلدى. تەپتىش قىلىشىمىزچە ئافاق خوجىنىڭ زىمىنلىرى قەشقەردە بولۇپ ئۇزۇن يىللاردىن بېرى ئۇنىڭ مەھسۇلاتلىرىنى قەشقەر ۋىلايەتلىك ئۇيغۇر ئۇيۇشمىسى تەسساۋۇپ قىلىپ كەلگەن، شۇنىڭ ئۈچۈن قەشقەردىكى ئۇيۇشما مەزكۈر كىشلەرنىڭ ئالالمىغان ھەقلىرىنى بىرشى لازىم...» بۇيرۇققا ئۆلكە رەئىسى لى يۇڭ تەستىق سالغانىدى.بۇ ‹‹بۇيرۇق ›› قەشقەر جەمئىيتىدە زور غۇلغۇلا پەيدا قىلدى، چۈنكى، ئۇلارنىڭ ئالىدىغىنى ھەربىرىنىڭ يىلىغا نەچچە ئون چارەك ئاش بولۇپ، ئون نەچچە كىشنىڭ 20 نەچچە يىلىغا ئالىدىغىنى بىر قانچە مىڭ چارەك ئاشلىقتىن ئىبارەت ئىدى.

بۇ ئاشلىقلار ھازىر بېرىش مۇمكىن ئەمەس ئىدى، چۈنكى، بۇنچە كۆپ ئاشلىق ئۇيۇشمىنىڭ ئامبىرىدا مەۋجۇت ئەمەس ئىدى. يىللاردىن بېرى ئافاق خوجىنىڭ نامىدىكى ۋەخپە زىمىنلار ھەركىم، ھەركىملەرنىڭ ئىلكىدە بولۇپ، چىققان ھوسۇلنى ئەنە شۇلار ئىشلىتىپ كەلگەن،ھازىر ئۇلارنىڭ كۆپىنچىسى يوق، بار بولغاندىمۇ ئۇلار ئۆز ۋاقتىدا ئىشلىتىپ كەتكەن، كىشلەر يەپ ئىچىپ، تۈگەتكەن، مۇنچىۋالا كۆپ ئاشلىقنى ئۇيۇشمىنىڭ قايتۇرۇپ بىرىشى مومكىن ئەمەس ئىدى. ئۇ ۋەخپە-زىمننى يىقىندىن بىرى ئۇيۇشما ئىگەللىگەن بولسىمۇ، ئۇيۇشما تەمىنلەپ كىلىۋاتقان مەكتەپ ئوقوغۇچىلىرى، دارىلئاجىزىن،دارىلتام، قىرى چۆرى، مىيىپ، يىتىملارنىڭ خىراجەتلىرى ئەنە شۇيەرنىڭ كىرىمىدىن بىرىلىپ كىلىۋاتاتتى ۋە بۇنىڭدىن كىيىنمۇ يىتىم يىسىرلارنى ئەنە شۇ كىرىمدىن تەمىنلەشكە توغرا كىلەتتى. مانا شۇ سەۋەپتىن قەشقەر جەمىيتى بۇ ھەقتەلەپ داۋاسىنى رەت قىلدى.

بۇ رەت قىلىش ئۈرۈمچى ھۆكۈمىتىنىڭ قارشىلىقىغا دۇچ كەلدى« قايتۇرۇش كىرەك» دىگەن بىسىم كۈچەيگەندى. دىگەن سۇئال قەشقەر خەلقىنىڭ يۈرەك ساداسى ئىدى. ئاففاق خوجىنىڭ ھەقتەلەپ دەۋاسى رەت قىلىنغاندىن كىيىن، بۇ ھەقتە قەشقەردە چىقىدىغان «يېڭى ھايات»گېزىتى ئافاق خوجىنىڭ تەرجىمىھالى، ئۇنىڭ ئۇيغۇر خەلقىگە كۆرسەتكەن« خىزمەتلىرى» كەلتۈرۈپ بەرگەن قىرغىنچىلىقلىرى، خەلق ئۇزۇن يىللاپ نادانلىققا، خوراپاتلىققا باشلىغان بىر«زات»ئىكەنلىكىنى ئاممىغا كەڭ تۈردە تونۇشتۇردى.

زىمىن مەسلىسىدە:ئافاق خوجىنىڭ زىمىننى ئەينى زاماندا زالىم ھۆكۈمەت بىلەن تېل بىرىكتۈرۈپ، خەلققە ئالۋاڭ-ياساق سىلىپ توپلىغان زىمىن ئىكەنلىكى، ھەر خىل ئالۋاڭ – ياساققا چىدىمىغان خەلق يەرلىرىنى تاشلاپ كەتكەنلىكىنى، ھەتتا بەزىلىرى ئافاق خوجىغا يەرلىرىنى سىلىق ھېساۋىغا بىكارغا بىرىش بىلەن ئالۋاڭدىن قۇتۇلغانلىقىنى، شۇ سەۋەپتىن نۇرغۇن يەرلەر ئافاق خوجىغا مەنسۇپ بولۇپ كەتكەنلىكىنى، ئافاق خوجىنىڭ مال-مۈلۈكى جۈملىدىن يەر-زىمىنلىرىنى خەلقتىن تارتىۋىلىش بەدىلىگە كەلگەنلىكىنى،ئۇنىڭ بۇ زىمىندا مىراسى يوقلىقىنى تونۇشتۇردى، بۇنى چۈشەنگەن خەلق: يەرلىكنىڭ نىمە ھەققى بار، ياقا يۇرتنىڭ،دىگەن«پەتۋا»نى چىقىرىپ ئافاق خوجىنىڭ ئەۋلادىغا ھېچنىمە بەرمەيمىز، دىيىشتى.بۇ ئىش ئافاق خوجا تەرەپدارلىرى بولغان بىر تۇركۇم ھوقۇقدارلارنىڭ ئوغىسىنى قايناتماي تۇرالمىدى. قەشقەردىكى ئافاق خوجىنى ‹‹پەيغەمبەر ئەۋلادى ››دەيدىغان شۇم ئغىزلار ‹‹خوجامنىڭ زىمىننى قايتۇرۇش كىرەك ›› ـــ دەپ جار سىلپ،مەخپى ھالدا ئۈرۈمچى ھۆكۈمىتىگە ئەرىز سۇنۇپ، قەشقەردىكى ئۇيغۇر ئۇيۇشمىسىنى تاقىۋېتىشنى ياكى ئۇنىڭ ھېساۋاتىنى تەكشۈرۈشنى، ئۈرۈمچىنىڭ ئۆتكۈزۋېلىپ باشقۇرشىنى تەلەپ قىلىشىپ، ھەقتەلەپ دەۋاسىنى ئاشكارە قوللىدى.

قەشقەردىكى ئۇيغۇر ئۇيۇشمىسى بۇ مىراس ئىتراپ قىلىنغاندا، ئۇيۇشما تەمىنلەيدىغان يىتىم – يىسىرلەر، ئىگە-چاقىسىز قالىدىغانلىقى توغۇرلۇق
قايتا-قايتا ھۆكۆمەتكە ئىلتىماس قىلغان بولسىمۇ مىراسخورلارنى يۆلەشنى ئاساس قىلغان ئۈرۈمچى ھۆكۈمىتى ئۇيۇشمىنىڭ مالىيىسىنى تەشكۈرۈپ كۆرىمىز ھەتتا ئۆتكۈزۋالىمىزدەپ بىسىم قىلغان بولسىمۇ بۇ دەۋا بىريىلغىچە سۆرۈلۈپ كەتتى، ئاخىرى ئۈرۈمچى «شىنجاڭ گېزىتى»باش مۇھەرىرى مەنسۇر ئەپەندى(ۋەتەن ئوغلى) باشچىلىقىدىكى ھېساپ تەكشۈرۈش كومىسىيىسى قەشقەرگە ئەۋەتىلدى، لېكىن ئۇلار قارشىلىققا ئۇچۇردى، زورلۇق قوللىنىش خەۋپى تۇغۇلغاندا بىرنەچچە قېتىم ئىچىلغان مەجلىسلەردە قەشقەرنىڭ قازىسى ئۆلىما ئابدۇغوپۇر داموللىنىڭ پاسسىپ ئورنىغا چۈشۈۋالغانلىقى ھەممىنىڭ غەزىپىنى قوزغىغانىدى، ئۇيغۇر ئۇيۇشمىسى ئۈرۈمچىدىن كەلگەن تەكشۈرۈش كومىسيىسى تەرپىدىن تەقىب ئاستىغا ئىلىنغانلىقتىن، ئەمدى قەشقەر خەلقى ئالدىدا بۇنىڭغا تاقابىل تۇرۇش كېرەكمۇقانداق؟ دېگەن سۇئال تۇغۇلدى، خەلقنىڭ قان –تەرىگە كەلگەن بۇ مەبلەغلەرنى ئۇلارغا ئۆتكۈزۈپ بىرىشنى ھېچكىم خالىمايتى، بۇھەقىتە ھەيئەت تەشكىل قىلدى، بىر قانچە قېتىملاپ يىغىنلار كومىسىيىسىگە ئەھۋال چۈشەندۈرۈلگەن بولسىمۇ لېكىن ئۇلار ئۆز گىپىدە چىڭ تۇرىۋەردى، سۈركىلىش كۈچەيدى، تالاش-تارتىشلار كۆپەيدى، ئاخىرى ئىش خەتەرلىك باسقۇچقا كىلىپ قالدى.

قەشقەر ۋىلايىتىنىڭ مۇئاۋىن گارنىزون قۇماندانى ئۇيغۇر ئاتاغلىق ئەسكىرى ئالتىنجى دىۋىزىيىنىڭ قۇماندانى ماخمۇت موھىتى يىغىنغا قاتنىشىپ خەلق تەرەپتە تۇرۇپ پىكىر قىلىش قارارىغا كەلدى:

-ئەپەندىلەر دېدى ، ئۇ قوشۇمىسىنى تۈرۈپ -؛ سىلەر ئۇيۇشمىمىزدىن راستىنلا ھېساپ ئالامسىلەر! ئۇيۇشمىمىز مالىيىسىدىكى توپلانغان مەبلەغ پۈتۈنلەي خەلىقنىڭ، ئۇ كىشلەر تەرپىدىن بىرىلگەن ئۆشۈر زاكات، ئىئانە، مازار ۋەخپە كىرمى قاتارلىقلاردىن كىرگەن. ئۇيۇشمىمىز بۇ كىرىم بىلەن نەچچە يۈزلىگەن قىرى، مىيىپ، ئىگە چاقىسىز كىشلەرنى ئاساس قىلغان دارىل ئاجىزىن، نەچچە يۈزلىگەن يىتىمنى ئاساس قىلغان دارىلېتاملارنى باقىدۇ، ئۇنىڭدىكى بالىلارنى ئوقوتىدۇ تەربىيەلەيدۇ، شەھەر ئەتراپىدىكى يۈزلىگەن مەكتەپلەردىكى ئوقوتقۇچىلار ئۇكىرىملەر بىلەن تەمىنلىنىدۇ، بۇھەقتە نەچچە قېتىملاپ چۈشەنچە بەرسەك قۇلاق سالمىدىڭلار سىلەرنىڭ كەلگۈسى ئەۋلاتلىرىمىزنىڭ، جەمىئەتنىڭ ئۈمۈدى، مىللەتنىڭ ئۈمۈدى، شۇنداقلا دۇنيانىڭ ئۈمۈدى بولغان ياش غۇنچىلارغا،ئاجىزلارغا سەرىپ قىلىدىغان بۇمەبلەغنى ئاشۇ مىراسخورلارغا ئىلىپ بىرىشنىڭلارغا قەتئى قارشى تۇرمەن؟! قەتئى قارشى تۇرمەن!

ئۇ يىغىندا سۈكۈت قىلىپ تۇرۇپ ئولتۇرۋالغان قازى رەئىس، كاتتا ئۆلما ئابدىغوپۇر داموللىغا نەپرەت كۆزى بىلەن تىكىلدى – دە ، پوزىتسىيىسىنى ئۆزگەرتتى:

-سىلەر يەنە ھېساپ ئالامسىلەر، ئەگەر شۇنداق قىلىدىغان بولساڭلار ھېسابنى مانا مۇشۇ بىرىدۇ؟!!...

مەخمۇت موھىيىتى بەتلەنگەن تاپانچىسىنى غەزەپ بىلەن شىرە ئۈستىگە تاشلىدى، ئۇنىڭ كۆزلىرىدىن غەزەپ ئۇچقۇنلىرى چاقناپ كەتتى. ئۇ ئۆز ئىرادىسىدىن ۋاز كىچىپ مەجلىسدىن چىقىپ كىتىۋاتقان «ھېساۋاتچىلارغا» مەغرۇرلۇق بىلەن قاراپ تۇراتتى.

مەخمۇت سىجاڭنىڭ بىۋاستە ئوتتۇرغا چېقىپ ئارلىشىشى ۋە خەلق تەرەپتە تۇرۇپ ھەققانىيەتنى ياقىلىشى، كەڭ خەلقنىڭ قاتتىق نارازىيىلىق بىلدۈرشى،ئۇرۇش مالىمانچىلىق تۈپەيلىدىن ئۇيغۇر ئۇيۇشما مالېىيىسىدە ۋەخپىنى تۆلىگۈدەك ئىقتىسادنىڭ بولمىغانلىق سەۋەپلىك تەپتىش كومىسىيىسى ئۈرۈمچىگە قايتىپ كەتتى، كىيىنكى ۋەقەلەرنىڭ تەرەققىيات داۋامىدا بۇ دەۋامۇ بىسىپ قالدى.
